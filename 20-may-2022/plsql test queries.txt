---TO PRINT FACT OF NUMBER--
DECLARE
I NUMBER;
N NUMBER;
RES NUMBER;
BEGIN
N :=6;
I :=1;
RES :=1;
FOR I IN 1..N LOOP
RES := RES*I;
END LOOP;
DBMS_OUTPUT.PUT_LINE(RES);
END;

----BY PASSING A STRING AND PRINT IT IN REVERSE ORDER--
CREATE OR REPLACE PROCEDURE STRING_S AS
A_S VARCHAR2(20);
BEGIN 
A_S := 'STATION';
FOR I IN REVERSE 1..LENGTH(A_S) LOOP
DBMS_OUTPUT.PUT_LINE(SUBSTR(A_S,I)); ----HERE I HAVE DOUBT---
END LOOP;
END;

---TO DISPLAY TOP 10 PRODUCTS BASED ON SALESAMOUNT--
SELECT * FROM PRODUCT_RETAIL;
SELECT * FROM SALES_RETAIL ;


SELECT *
            FROM (SELECT P_NAME, dense_rank() OVER ORDER BY SUM(AMOUNT)DESC)R
            FROM PRODUCT_RETAIL P,SALES_RETAIL S
            WHERE P.PROD_ID = S.PROD_ID;
            GROUP BY P.P_NAME)C
WHERE R<10;
